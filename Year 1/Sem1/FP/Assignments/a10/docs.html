18        """
19        Set up the game: assign colors and initialize players.

21        self.human_player = Player(self.board, player_color)

27    def play_turn(self):
28        """
29        Handle a single turn for the current player.
30        """
31        if isinstance(self.current_player, Player):
32            # Human player's turn
33            while True:
34                try:
35                    column = int(input(f"{Color.to_string(self.current_player.color)}, enter the column (0-6) to drop your disc: "))
36                    self.board.place_disc(column, self.current_player.color)
37                    break  # Exit the loop once a valid move is made
38                except (OutOfBoundsExceptions, ColumnFilled) as e:
39                    print(f"Error: {e}. Try again.")
40                except ValueError:
41                    print("Invalid input. Please enter a number between 0 and 6.")
42        else:
43            # Computer player's turn
44            if self.difficulty == 1:
45                column = self.computer_player.easy_difficulty()
46            elif self.difficulty == 2:
47                column = self.computer_player.medium_difficulty(self.human_player.color)
48            elif self.difficulty == 3:
49                column = self.computer_player.hard_difficulty()
50
51            self.board.place_disc(column, self.computer_player.color)
52
53    def switch_player(self):
54        """
55        Switch the current player between the human player and the computer player.
56        """
57        self.current_player = self.computer_player if self.current_player == self.human_player else self.human_player

59    def check_game_over(self):
60        """
61        Check if the game has ended with a win or a draw.
62        """
63        if self.board.check_victory(self.current_player.color):
64            print(self.board)
65            if isinstance(self.current_player, Player):
66                print("Congratulations! You won! :)")
67            else:
68                print("The computer won. Better luck next time!")
69            self.is_running = False
70            return
71
72        if self.board.is_full():

74            print("The game is a draw!")
75            self.is_running = False

78        """
79        Check if the game is a draw.
80        """
81        return self.board.is_full() and not self.board.check_victory(self.human_player.color) and not self.board.check_victory(self.computer_player.color)

A class whose instances are single test cases.

By default, the test code itself should be placed in a method named 'runTest'.

If the fixture may be used for many test cases, create as many test methods as are needed. When instantiating such a TestCase subclass, specify in the constructor arguments the name of the test method that the instance is to execute.

Test authors should subclass TestCase for their own tests. Construction and deconstruction of the test's environment ('fixture') can be implemented by overriding the 'setUp' and 'tearDown' methods respectively.

If it is necessary to override the __init__ method, the base class __init__ method must always be called. It is important that subclasses should not change the signature of their __init__ method, since instances of the classes are instantiated automatically by parts of the framework in order to be run.

When subclassing TestCase, you can set these attributes:

  • failureException: determines which exception will be raised when the instance's assertion methods fail; test methods raising this exception will be deemed to have 'failed' rather than 'errored'.
  • longMessage: determines whether long messages (including repr of objects used in assert methods) will be printed on failure in addition to any explicit message passed.
  • maxDiff: sets the maximum length of a diff in failure messages by assert methods using difflib. It is looked up as an instance attribute so can be configured by individual tests if required.

 68    def get_player_preferences(self):
 69        """
 70        Get player color and difficulty from the user.

 72        print("Choose your color: 1-Red, 2-Green, 3-Blue, 4-Pink, 5-Yellow")
 73        color_choice = int(input("Enter the number corresponding to your color choice: "))
 74        player_color = COLOR_OPTIONS[color_choice]
 75
 76        print("Choose difficulty: 1-Easy, 2-Medium, 3-Hard")
 77        difficulty = int(input("Enter the number corresponding to your difficulty choice: "))

 79
 80    def run(self):
 81        """
 82        Main method to run the game.
 83        """
 84        self.display_welcome()
 85        player_color, difficulty = self.get_player_preferences()
 86        self._game.setup_game(player_color, difficulty)
 87
 88        while self._game.is_running:
 89            self.display_board()
 90            self.display_turn_prompt()
 91            self._game.play_turn()
 92            self._game.check_game_over()
 93
 94            if self._game.is_running:
 95                self._game.switch_player()
 96
 97        # Game over state
 98        self.display_board()
 99        if self._game.check_draw():
100            self.display_draw()
101        else:
102            self.display_winner()

241                            pygame.mixer.Sound.play(win_sound)
242                        else:
243                            game.switch_player()
244
245        if game.current_player == game.computer_player and not game.game_over:
246            pygame.time.wait(500)  # Simulate thinking
247            game.computer_turn()
248            if game.check_game_over():

261            )

263            pygame.display.update()
